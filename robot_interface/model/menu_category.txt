    def create_dict(self):
        my_dict = {}
        for i in range(len(self.area)):
            if i < len(self.valid_urls):
                for url in self.valid_urls:
                    # replace dashes with spaces for case-insensitive matching
                    area_name = self.area[i].replace('-', ' ').lower()
                    url_name = url.split('/')[-1].replace('-', ' ').lower()
                    if area_name in url_name:
                        my_dict[self.area[i]] = url
                        break
                else:
                    my_dict[self.area[i]] = None
            else:
                my_dict[self.area[i]] = None
        return my_dict
